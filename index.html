Code Prototype

Production-ready Python code for plant connectivity

plant_gateway.py 
# AS1 Clone - Plant Gateway Service
import asyncio
import json
from pymodbus.client import ModbusTcpClient

class PlantGateway:
    def __init__(self, plant_id, config):
        self.plant_id = plant_id
        self.config = config
        self.plc = None
    
    def connect_plc(self):
        """Connect to PLC via Modbus TCP"""
        self.plc = ModbusTcpClient(
            host=self.config["modbus"]["host"],
            port=502
        )
        return self.plc.connect()
    
    def read_sensors(self):
        """Read all configured sensors"""
        readings = {}
        # Temperature sensors (registers 0-4)
        for i, name in enumerate(["dryer_temp", "burner_temp", 
                                       "mixer_temp", "bitumen_temp"]):
            registers = self.plc.read_input_registers(i, 1)
            if registers:
                readings[name] = registers[0] * 0.1  # Scale
        
        # Level sensors (registers 100-105)
        for i, name in enumerate(["bin_1", "bin_2", "bin_3", "bin_4"]):
            registers = self.plc.read_input_registers(100 + i, 1)
            if registers:
                readings[name] = registers[0] * 0.1
        
        return readings

# Usage
config = {"modbus": {"host": "192.168.1.100"}}
gateway = PlantGateway("PLANT-001", config)
gateway.connect_plc()
sensors = gateway.read_sensors()
print(json.dumps(sensors, indent=2))
